根据`Promise.all()`的原理扩展一下显示多个请求的进度信息

<!-- more -->

```typescript
import { Observable } from "rxjs";
import { delay } from "rxjs/operators";

function delayValue(value: any, delayTime: number): Observable<any> {
  return new Observable(subscriber => {
    subscriber.next(value);
    subscriber.complete();
  }).pipe(delay(delayTime));
}

function showProgress(
  observables: Array<Observable<any>> = [],
  progressCallback: (progress: number) => void
): Observable<Array<any>> {
  let count = 0;
  const result: any = [];
  return new Observable(subscriber => {
    if (observables.length) {
      observables.forEach((item, index) => {
        count++;
        item.subscribe(value => {
          count--;
          result[index] = value;
          const progress =
            result.filter((v: any) => v).length / observables.length;
          progressCallback(progress);
          if (!count) {
            subscriber.next(result);
            subscriber.complete();
          }
        });
      });
    } else {
      progressCallback(1);
      subscriber.next(result);
      subscriber.complete();
    }
  });
}

showProgress(
  [
    delayValue(1, 1000),
    delayValue(2, 500),
    delayValue(3, 2000),
    delayValue(4, 1500)
  ],
  // 显示进度信息
  progress => {
    console.log("Progress: " + progress);
  }
).subscribe(value => console.log(value));

// Progress: 0.25
// Progress: 0.5
// Progress: 0.75
// Progress: 1
// [ 1, 2, 3, 4 ]
```
